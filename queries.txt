CREATE TABLE tblproduct (
 id int NOT NULL,
 active varchar(100) DEFAULT NULL,
 code varchar(5) DEFAULT NULL,
 create_date date default sysdate not null,
 description varchar(255) DEFAULT NULL,
 image varchar(100) DEFAULT NULL,
 image_name varchar(400) DEFAULT NULL,
 name varchar(30) DEFAULT NULL,
 price varchar(200) DEFAULT NULL,
 mrp_price varchar(200) DEFAULT NULL,
 product_category varchar(100) DEFAULT NULL,
 PRIMARY KEY (id)
) ;




CREATE SEQUENCE SEQ_tblproduct;





CREATE OR REPLACE TRIGGER TRG_T1
BEFORE INSERT ON tblproduct
FOR EACH ROW
BEGIN
    SELECT SEQ_tblproduct.NEXTVAL
    INTO :NEW.id
    FROM DUAL;
END;
/
